  def guess_format_and_run(confpath)
    case File.extname(confpath)
    when /\.xml/i
      run_with_xml(confpath)
    when /\.yaml/i
      run_with_yaml(confpath)
    else
      raise "configfile(#{confpath})'s suffix is not yaml nor xml.use -x or -y option."
    end
  end
