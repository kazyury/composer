    def fullname()
      wk=@wkstation
      jo=@stream
      if wk && jo
        return wk+'#'+jo
      elsif jo
        return jo
      else
        if block_given?
          val=yield
        else
          val='_NO_ENTRY_'
        end
        @logger.warn("#{self.class}"){"no name is specified. use [ #{val} ] instead."}
        return val
      end
    end
