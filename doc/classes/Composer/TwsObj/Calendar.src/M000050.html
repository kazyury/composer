    def append_date(mmddyy)
      @date.push mmddyy
      if /(\d\d)\/(\d\d)\/(\d\d)/=~mmddyy
        @date_internal.push Date.parse(mmddyy,true)
        @date_internal.sort!
      else
        @logger.warn("#{self.class}"){ "defined date [ #{mmddyy} } may not be parsed correctly." }
      end
    end
