    def to_s
      ret=''
      ret << fullname if @name
      if @scriptname
        ret << ' SCRIPTNAME ' << @scriptname
      elsif @docommand
        ret << ' DOCOMMAND ' << @docommand
      end
      ret << ' STREAMLOGON ' << @streamlogon if @streamlogon
      ret
    end
