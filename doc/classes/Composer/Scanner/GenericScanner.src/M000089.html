    def reserved_word(case_sensitive,*words)
      words.each{|word|
        if case_sensitive
          reg=Regexp.new('\A('+word+')[^\w-]')
        else
          reg=Regexp.new('\A('+word+')[^\w-]',Regexp::IGNORECASE)
        end
        @tokens.push [reg, word.upcase.intern, true]
      }
    end
