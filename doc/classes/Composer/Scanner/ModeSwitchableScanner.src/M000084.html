    def builtin_token(token)
      @tokens=@mode_frame[@mode]
      super
      if @mode==:DEFAULT
        if token==:NUMBER
          regist_default_token(BUILTIN_TOKENS[token],token,true)
        else
          regist_default_token(BUILTIN_TOKENS[token],token,false)
        end
      end
    end
