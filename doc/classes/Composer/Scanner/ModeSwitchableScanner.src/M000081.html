    def on(mode=:DEFAULT)
      @mode=mode
      @tokens=@mode_frame[@mode] # 指定されたフレームをカレントに
      if @tokens.empty?
        @mode_frame[mode]=@mode_frame[:DEFAULT].dup
        @tokens=@mode_frame[@mode]
      end
      yield self
      @mode_frame.merge!({ mode => @tokens })
      @mode=:DEFAULT             # 抜けるときは、:DEFAULTに戻す。
      @tokens=@mode_frame[@mode]
    end
